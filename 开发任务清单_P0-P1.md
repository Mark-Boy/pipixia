### T001 — 用户认证模块（JWT Login/Register）

#### 1.1 需求说明
- 用户注册/登录，生成 Access Token（2h）+ Refresh Token（7d）
- Refresh Token 轮换（Token Rotation）防重放
- Token 黑名单（登出/过期）
- Cookie 安全属性（HttpOnly + Secure + SameSite）

#### 1.2 数据库模型（`models/user.py`）

#### 1.4 路由实现（`routers/auth.py`）

```python
from fastapi import APIRouter, Depends, HTTPException, status, Response
from sqlalchemy.orm import Session
from api.db.session import get_db
from api.schemas.auth import UserCreate, LoginRequest, TokenResponse, RefreshTokenRequest
from api.services.auth import (
    create_access_token,
    create_refresh_token,
    verify_password,
    get_password_hash,
    blacklist_token,
    is_token_blacklisted,
    get_current_user,
    refresh_tokens
)
from api.models.user import User
from typing import Annotated
import logging

logger = logging.getLogger(__name__)
router = APIRouter(prefix="/api/auth", tags=["认证"])

@router.post("/register", response_model=TokenResponse, status_code=201)
async def register(user_data: UserCreate, db: Session = Depends(get_db), response: Response = None):
    """用户注册（创建用户 + 颁发 Token）"""
    # 检查用户名/邮箱是否存在
    if db.query(User).filter(User.username == user_data.username).first():
        raise HTTPException(409, "用户名已存在")
    if db.query(User).filter(User.email == user_data.email).first():
        raise HTTPException(409, "邮箱已被注册")
    
    # 创建用户
    new_user = User(
        username=user_data.username,
        email=user_data.email,
        password_hash=get_password_hash(user_data.password),
        role=user_data.role,
        is_active=True
    )
    db.add(new_user)
    db.commit()
    db.refresh(new_user)
    
    # 生成 Token
    access_token = create_access_token(user_id=str(new_user.id), role=new_user.role)
    refresh_token = create_refresh_token(user_id=str(new_user.id))
    
    # 设置 Cookie（HttpOnly + Secure + SameSite）
    response.set_cookie(
        key="refresh_token",
        value=refresh_token,
        httponly=True,
        secure=True,
        samesite="lax",
        max_age=604800,  # 7d
        path="/api/v1"
    )
    
    logger.info(f"用户注册成功: {new_user.username}")
    return TokenResponse(access_token=access_token, refresh_token=refresh_token)

@router.post("/login", response_model=TokenResponse)
async def login(login_data: LoginRequest, db: Session = Depends(get_db), response: Response = None):
    """用户登录"""
    user = db.query(User).filter(User.username == login_data.username).first()
    if not user or not verify_password(login_data.password, user.password_hash):
        raise HTTPException(401, "用户名或密码错误")
    
    if not user.is_active:
        raise HTTPException(403, "账号已禁用")
    
    # 更新最后登录时间
    from datetime import datetime
    user.last_login_at = datetime.utcnow()
    db.commit()
    
    access_token = create_access_token(user_id=str(user.id), role=user.role)
    refresh_token = create_refresh_token(user_id=str(user.id))
    
    response.set_cookie("refresh_token", refresh_token, httponly=True, secure=True, samesite="lax", max_age=604800, path="/api/v1")
    logger.info(f"用户登录: {user.username}")
    return TokenResponse(access_token=access_token, refresh_token=refresh_token)

@router.post("/refresh", response_model=TokenResponse)
async def refresh_token(req: RefreshTokenRequest, db: Session = Depends(get_db)):
    """刷新 Token（Token Rotation）"""
    if is_token_blacklisted(req.refresh_token):
        raise HTTPException(401, "Token 已失效")
    
    new_access, new_refresh = refresh_tokens(req.refresh_token, db)
    return TokenResponse(access_token=new_access, refresh_token=new_refresh)

@router.post("/logout")
async def logout(response: Response, db: Session = Depends(get_db), current_user: User = Depends(get_current_user)):
    """用户登出（Token 黑名单）"""
    from fastapi import Header
    current_token = Header(default=None)
    if current_token:
        blacklist_token(current_token)
    
    # 清除 Cookie
    response.set_cookie("refresh_token", "", httponly=True, secure=True, samesite="lax", max_age=0, path="/api/v1")
    logger.info(f"用户登出: {current_user.username}")
    return {"message": "登出成功"}
```

#### 1.5 安全中间件（`middleware/auth.py`）

```python
from fastapi import Depends, HTTPException, status
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
from jose import JWTError, jwt
from sqlalchemy.orm import Session
from api.db.session import get_db
from api.models.user import User
from api.config import settings
import logging

logger = logging.getLogger(__name__)
security = HTTPBearer()

# Token 黑名单（Redis）
from api.services.cache import get_cache, set_cache

def get_current_user(credentials: HTTPAuthorizationCredentials = Depends(security), db: Session = Depends(get_db)):
    """JWT 鉴权中间件"""
    token = credentials.credentials
    
    # 检查黑名单
    if get_cache(f"blacklist:{token}"):
        raise HTTPException(401, "Token 已失效")
    
    try:
        payload = jwt.decode(token, settings.SECRET_KEY, algorithms=[settings.ALGORITHM])
        user_id = payload.get("sub")
        role = payload.get("role")
        if user_id is None:
            raise HTTPException(401, "Token 无效")
    except JWTError:
        raise HTTPException(401, "Token 格式错误")
    
    user = db.query(User).filter(User.id == user_id).first()
    if not user or not user.is_active:
        raise HTTPException(403, "用户不存在或已禁用")
    
    return user
```

### T010 — 店铺管理 API（CRUD + Token 加密）

#### 1.1 需求说明
- 店铺 CRUD（增删改查）

#### 1.2 路由实现（`routers/shops.py`）

```python
from fastapi import APIRouter, Depends, HTTPException, status, Query
from sqlalchemy.orm import Session
from sqlalchemy import func
from api.db.session import get_db
from api.schemas.shop import ShopCreate, ShopUpdate, ShopResponse
from api.models.shop import Shop
from api.models.user import User
from api.services.crypto import encrypt_aes256, decrypt_aes256
from typing import Annotated, Optional, List
import logging

logger = logging.getLogger(__name__)
router = APIRouter(prefix="/api/shops", tags=["店铺管理"])

@router.get("", response_model=List[ShopResponse])
async def get_shops(
    page: int = Query(1, ge=1),
    size: int = Query(20, ge=1, le=100),
    active: Optional[bool] = Query(None),
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """店铺列表（自动按 shop_id 过滤）"""
    query = db.query(Shop).filter(Shop.user_id == current_user.id)
    
    if active is not None:
        query = query.filter(Shop.is_active == active)
    
    # 分页
    total = query.count()
    shops = query.offset((page - 1) * size).limit(size).all()
    
    return [ShopResponse.model_validate(s) for s in shops]

@router.post("", response_model=ShopResponse, status_code=201)
async def create_shop(
    shop_data: ShopCreate,
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """创建店铺（Token 加密存储）"""
    # 检查店铺名是否已存在
    existing = db.query(Shop).filter(
        Shop.user_id == current_user.id,
        Shop.shop_name == shop_data.shop_name
    ).first()
    if existing:
        raise HTTPException(409, "店铺名已存在")
    
    # 加密 Token
    encrypted_token = encrypt_aes256(shop_data.shop_token)
    
    new_shop = Shop(
        user_id=current_user.id,
        shop_name=shop_data.shop_name,
        platform=shop_data.platform,
        shop_token_encrypted=encrypted_token,
        config_json=shop_data.config_json or {},
        is_active=True
    )
    db.add(new_shop)
    db.commit()
    db.refresh(new_shop)
    
    logger.info(f"店铺创建: {shop_data.shop_name}")
    return ShopResponse.model_validate(new_shop)

@router.put("/{shop_id}", response_model=ShopResponse)
async def update_shop(
    shop_id: str,
    shop_data: ShopUpdate,
    db: Session = Depends(get_db),

        shop.shop_name = shop_data.shop_name
